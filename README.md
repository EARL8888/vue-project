## 使用
环境 node 10.x

## 使用
``` json
git clone https://github.com/EARL8888/vue-project.git

npm install
```

## 项目 目录

```json 
├── node_modules               // node 命令行构建工具 
├── build                      // 构建相关  
├── config                     // 配置相关
├── src
│   ├── api                    // view界面
│   │    ├── api               // 所有请求
│   ├── components             // 全局UI组件
│   ├── config                 // 网络配置文件
│   │    ├── http              // 网络拦截器
│   │    ├──url                // 域名
│   ├── router                 // 路由
│   ├── store                  // 全局store管理
│   ├── utils                  // 全局公用方法
│   ├── page                   // view界面
│   ├── App.vue                // 入口页面
│   └── main.js                // 入口 加载组件 初始化等
├── static                     // 静态资源
│   ├── img                    // img
│   ├── css                    // css
│   ├── js                     // js
├── .babelrc                   // babel-loader 配置
├── eslintrc.js                // eslint 配置项
├── .gitignore                 // git 忽略项
├── favicon.ico                // favicon图标
├── index.html                 // html模板
└── package.json               // package.json

```
